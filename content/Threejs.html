Threejs

import two stl parts and use mouse to move part around:

const scene = new THREE.Scene();
const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);

const renderer = new THREE.WebGLRenderer();
renderer.setSize(window.innerWidth, window.innerHeight);
document.body.appendChild(renderer.domElement);

const controls = new THREE.OrbitControls(camera, renderer.domElement);

const loader = new THREE.STLLoader();

let mesh1, mesh2;

loader.load('part1.stl', function (geometry) {
    const material = new THREE.MeshBasicMaterial({color: 0x00ff00});
    mesh1 = new THREE.Mesh(geometry, material);
    scene.add(mesh1);
});

loader.load('part2.stl', function (geometry) {
    const material = new THREE.MeshBasicMaterial({color: 0xff0000});
    mesh2 = new THREE.Mesh(geometry, material);
    scene.add(mesh2);
});

camera.position.z = 5;

const dragControls = new THREE.DragControls([mesh1, mesh2], camera, renderer.domElement);
dragControls.addEventListener('dragstart', function (event) {
    controls.enabled = false;
});
dragControls.addEventListener('dragend', function (event) {
    controls.enabled = true;
});

function animate() {
    requestAnimationFrame(animate);
    renderer.render(scene, camera);
}
animate();

use Brython to rewrite the above program:

from browser import document, window

THREE = window.THREE

scene = THREE.Scene.new()
camera = THREE.PerspectiveCamera.new(75, window.innerWidth / window.innerHeight, 0.1, 1000)

renderer = THREE.WebGLRenderer.new()
renderer.setSize(window.innerWidth, window.innerHeight)
document.body.appendChild(renderer.domElement)

controls = THREE.OrbitControls.new(camera, renderer.domElement)

loader = THREE.STLLoader.new()

mesh1 = None
mesh2 = None

def onPart1Load(geometry):
    global mesh1
    material = THREE.MeshBasicMaterial.new({"color": 0x00ff00})
    mesh1 = THREE.Mesh.new(geometry, material)
    scene.add(mesh1)

loader.load('part1.stl', onPart1Load)

def onPart2Load(geometry):
    global mesh2
    material = THREE.MeshBasicMaterial.new({"color": 0xff0000})
    mesh2 = THREE.Mesh.new(geometry, material)
    scene.add(mesh2)

loader.load('part2.stl', onPart2Load)

camera.position.z = 5

dragControls = THREE.DragControls.new([mesh1, mesh2], camera, renderer.domElement)

def onDragStart(event):
    controls.enabled = False

dragControls.addEventListener('dragstart', onDragStart)

def onDragEnd(event):
    controls.enabled = True

dragControls.addEventListener('dragend', onDragEnd)

def animate():
    window.requestAnimationFrame(animate)
    renderer.render(scene, camera)

animate()
